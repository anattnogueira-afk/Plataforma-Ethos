import { useState, useEffect } from 'react';
import { X, Star, Send, Mail } from 'lucide-react';

interface FeedbackModalProps {
  questionNumber: number;
  onClose: () => void;
}

export function FeedbackModal({ questionNumber, onClose }: FeedbackModalProps) {
  const [rating, setRating] = useState<number>(0);
  const [hoveredRating, setHoveredRating] = useState<number>(0);
  const [objective, setObjective] = useState<string>('');
  const [improvements, setImprovements] = useState<string[]>([]);
  const [newFeature, setNewFeature] = useState('');
  const [email, setEmail] = useState('');

  // Gerencia foco ao abrir modal
  useEffect(() => {
    const previouslyFocusedElement = document.activeElement as HTMLElement;
    
    return () => {
      // Retorna foco ao fechar
      if (previouslyFocusedElement) {
        previouslyFocusedElement.focus();
      }
    };
  }, []);

  const objectiveOptions = [
    'Entender dilemas éticos em telemedicina',
    'Avaliar minha prática profissional',
    'Aprender sobre regulamentações',
    'Outro (por favor, especifique)'
  ];

  const improvementOptions = [
    'Mais perguntas contextuais',
    'Explicações mais detalhadas',
    'Interface mais intuitiva',
    'Outro (especifique)'
  ];

  const handleObjectiveChange = (value: string) => {
    setObjective(value);
  };

  const handleImprovementToggle = (value: string) => {
    if (improvements.includes(value)) {
      setImprovements(improvements.filter(item => item !== value));
    } else {
      setImprovements([...improvements, value]);
    }
  };

  const handleSubmit = () => {
    console.log('Feedback:', { 
      questionNumber, 
      rating, 
      objective, 
      improvements,
      newFeature,
      email 
    });
    onClose();
  };

  const handleSkip = () => {
    onClose();
  };

  // Fecha modal ao pressionar Escape
  useEffect(() => {
    const handleEscape = (event: KeyboardEvent) => {
      if (event.key === 'Escape') {
        onClose();
      }
    };

    document.addEventListener('keydown', handleEscape);
    return () => document.removeEventListener('keydown', handleEscape);
  }, [onClose]);

  return (
    <div 
      className="fixed inset-0 z-[100] flex items-center justify-center p-4" 
      style={{ backgroundColor: 'rgba(0, 0, 0, 0.6)' }}
      role="dialog"
      aria-modal="true"
      aria-labelledby="feedback-modal-title"
      onClick={onClose}
    >
      {/* Modal */}
      <div 
        className="relative bg-popover max-w-2xl w-full p-6 sm:p-8 max-h-[90vh] overflow-y-auto"
        style={{ 
          borderRadius: 'calc(var(--radius) + 8px)',
          boxShadow: '0 25px 50px -12px rgba(0, 0, 0, 0.25)'
        }}
        onClick={(e) => e.stopPropagation()}
      >
        <button
          onClick={handleSkip}
          className="absolute top-4 sm:top-6 right-4 sm:right-6 p-2 hover:bg-secondary transition-colors rounded-full focus:outline-none focus:ring-2 focus:ring-ring"
          aria-label="Fechar modal de feedback"
        >
          <X size={20} aria-hidden="true" />
        </button>

        <h2 id="feedback-modal-title" className="mb-6 sm:mb-8 text-primary pr-8">Sua Contribuição é Valiosa!</h2>
        
        {/* Rating Question */}
        <div className="mb-6 sm:mb-8">
          <label className="block mb-4" style={{ fontWeight: 'var(--font-weight-medium)', fontSize: '16px' }}>
            Como você avalia sua experiência geral com o simulador?
          </label>
          <div className="flex gap-2 sm:gap-3 flex-wrap" role="radiogroup" aria-label="Avaliação de 1 a 5 estrelas">
            {[1, 2, 3, 4, 5].map((star) => (
              <button
                key={star}
                type="button"
                onClick={() => setRating(star)}
                onMouseEnter={() => setHoveredRating(star)}
                onMouseLeave={() => setHoveredRating(0)}
                className="transition-transform hover:scale-110 focus:outline-none focus:ring-2 focus:ring-ring rounded"
                aria-label={`${star} ${star === 1 ? 'estrela' : 'estrelas'}`}
                aria-checked={rating === star}
                role="radio"
              >
                <Star
                  size={36}
                  className={
                    star <= (hoveredRating || rating)
                      ? 'fill-primary stroke-primary'
                      : 'stroke-border'
                  }
                  aria-hidden="true"
                />
              </button>
            ))}
          </div>
        </div>

        {/* Objective Question */}
        <fieldset className="mb-6 sm:mb-8">
          <legend className="mb-4" style={{ fontWeight: 'var(--font-weight-medium)', fontSize: '16px' }}>
            Qual é seu principal objetivo ao utilizar o Ethos? (Selecione uma opção)
          </legend>
          <div className="grid grid-cols-1 sm:grid-cols-2 gap-3 sm:gap-4">
            {objectiveOptions.map((option) => (
              <label
                key={option}
                className={`flex items-start gap-3 p-3 sm:p-4 border cursor-pointer transition-colors focus-within:ring-2 focus-within:ring-ring ${
                  objective === option
                    ? 'border-primary bg-primary/5'
                    : 'border-border hover:border-primary/50'
                }`}
                style={{ borderRadius: 'var(--radius)' }}
              >
                <input
                  type="radio"
                  name="objective"
                  value={option}
                  checked={objective === option}
                  onChange={(e) => handleObjectiveChange(e.target.value)}
                  className="mt-0.5 focus:ring-2 focus:ring-ring flex-shrink-0"
                />
                <span style={{ fontSize: '16px' }}>{option}</span>
              </label>
            ))}
          </div>
        </fieldset>

        {/* Improvements Question */}
        <fieldset className="mb-6 sm:mb-8">
          <legend className="mb-4" style={{ fontWeight: 'var(--font-weight-medium)', fontSize: '16px' }}>
            Quais áreas necessitam melhorias? (Selecione todas as que se aplicam)
          </legend>
          <div className="grid grid-cols-1 sm:grid-cols-2 gap-3 sm:gap-4">
            {improvementOptions.map((option) => (
              <label
                key={option}
                className={`flex items-start gap-3 p-3 sm:p-4 border cursor-pointer transition-colors focus-within:ring-2 focus-within:ring-ring ${
                  improvements.includes(option)
                    ? 'border-primary bg-primary/5'
                    : 'border-border hover:border-primary/50'
                }`}
                style={{ borderRadius: 'var(--radius)' }}
              >
                <input
                  type="checkbox"
                  value={option}
                  checked={improvements.includes(option)}
                  onChange={() => handleImprovementToggle(option)}
                  className="mt-0.5 focus:ring-2 focus:ring-ring flex-shrink-0"
                />
                <span style={{ fontSize: '16px' }}>{option}</span>
              </label>
            ))}
          </div>
        </fieldset>

        {/* New Feature Question */}
        <div className="mb-6 sm:mb-8">
          <label htmlFor="new-feature" className="block mb-3" style={{ fontWeight: 'var(--font-weight-medium)', fontSize: '16px' }}>
            Que novo recurso você gostaria de implementar no Ethos?
          </label>
          <textarea
            id="new-feature"
            value={newFeature}
            onChange={(e) => setNewFeature(e.target.value)}
            placeholder="Descreva o novo recurso que você gostaria de implementar..."
            rows={4}
            className="w-full p-3 sm:p-4 border border-border bg-input-background focus:border-primary focus:ring-2 focus:ring-ring outline-none transition-all resize-none"
            style={{ borderRadius: 'var(--radius)', fontSize: '16px' }}
            aria-describedby="new-feature-description"
          />
          <span id="new-feature-description" className="sr-only">Campo de texto para sugestões de novos recursos</span>
        </div>

        {/* Email Input */}
        <div className="mb-6 sm:mb-8">
          <label htmlFor="email-input" className="block mb-3" style={{ fontWeight: 'var(--font-weight-medium)', fontSize: '16px' }}>
            E-mail para contato (opcional)
          </label>
          <div className="relative">
            <Mail className="absolute left-3 sm:left-4 top-1/2 -translate-y-1/2 text-muted-foreground" size={20} aria-hidden="true" />
            <input
              id="email-input"
              type="email"
              value={email}
              onChange={(e) => setEmail(e.target.value)}
              placeholder="seu.email@empresa.com"
              className="w-full pl-11 sm:pl-12 pr-4 py-3 sm:py-4 border border-border bg-input-background focus:border-primary focus:ring-2 focus:ring-ring outline-none transition-all"
              style={{ borderRadius: 'var(--radius)', fontSize: '16px' }}
              aria-describedby="email-description"
            />
            <span id="email-description" className="sr-only">Digite seu e-mail caso queira ser contatado</span>
          </div>
        </div>

        {/* Actions */}
        <div className="flex flex-col sm:flex-row gap-3 sm:gap-4">
          <button
            type="button"
            onClick={handleSkip}
            className="flex-1 py-3 sm:py-4 border border-border hover:bg-secondary transition-colors focus:outline-none focus:ring-2 focus:ring-ring"
            style={{ borderRadius: 'var(--radius)', fontSize: '16px' }}
            aria-label="Cancelar feedback"
          >
            Cancelar
          </button>
          <button
            type="button"
            onClick={handleSubmit}
            disabled={rating === 0}
            className={`flex-1 py-3 sm:py-4 transition-all flex items-center justify-center gap-2 focus:outline-none focus:ring-2 focus:ring-ring ${
              rating > 0
                ? 'bg-primary text-primary-foreground hover:opacity-90'
                : 'bg-muted text-muted-foreground cursor-not-allowed'
            }`}
            style={{ borderRadius: 'var(--radius)', fontSize: '16px' }}
            aria-disabled={rating === 0}
            aria-label={rating === 0 ? 'Enviar contribuição (avaliação necessária)' : 'Enviar contribuição'}
          >
            <Send size={18} aria-hidden="true" />
            Enviar Contribuição
          </button>
        </div>
      </div>
    </div>
  );
}
