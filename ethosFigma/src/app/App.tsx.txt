import { useState } from 'react';
import { Header } from './components/Header';
import { Hero } from './components/Hero';
import { QuestionnaireSection } from './components/QuestionnaireSection';
import { ResultsSection } from './components/ResultsSection';
import { Footer } from './components/Footer';
import { FloatingFeedbackButton } from './components/FloatingFeedbackButton';
import { FeedbackModal } from './components/FeedbackModal';

export type ImpactLevel = 'risco-regulatorio' | 'uso-aceitavel' | 'uso-cuidado' | 'nao-sensivel' | null;

export default function App() {
  const [showQuestionnaire, setShowQuestionnaire] = useState(false);
  const [quizCompleted, setQuizCompleted] = useState(false);
  const [impactLevel, setImpactLevel] = useState<ImpactLevel>(null);
  const [showFeedbackModal, setShowFeedbackModal] = useState(false);

  const scrollToSection = (sectionId: string) => {
    const element = document.getElementById(sectionId);
    element?.scrollIntoView({ behavior: 'smooth' });
  };

  const handleStartQuiz = () => {
    setShowQuestionnaire(true);
    setQuizCompleted(false);
    setImpactLevel(null);
    setTimeout(() => scrollToSection('questionnaire'), 100);
  };

  const handleQuizComplete = (calculatedImpact: ImpactLevel) => {
    setImpactLevel(calculatedImpact);
    setQuizCompleted(true);
  };

  const handleRestart = () => {
    setShowQuestionnaire(false);
    setQuizCompleted(false);
    setImpactLevel(null);
    window.scrollTo({ top: 0, behavior: 'smooth' });
  };

  const handleOpenFeedback = () => {
    setShowFeedbackModal(true);
  };

  const handleCloseFeedback = () => {
    setShowFeedbackModal(false);
  };

  return (
    <div className="min-h-screen flex flex-col">
      {/* Skip to main content link */}
      <a 
        href="#main-content" 
        className="sr-only focus:not-sr-only focus:absolute focus:top-4 focus:left-4 focus:z-[200] focus:px-6 focus:py-3 focus:bg-primary focus:text-primary-foreground focus:outline-none focus:ring-4 focus:ring-ring"
        style={{ borderRadius: 'var(--radius)' }}
      >
        Pular para o conteúdo principal
      </a>

      <Header onNavigate={scrollToSection} onStartQuiz={handleStartQuiz} />
      
      <main id="main-content" className="flex-1" role="main">
        <Hero onStartQuiz={handleStartQuiz} />
        
        {showQuestionnaire && !quizCompleted && (
          <QuestionnaireSection onComplete={handleQuizComplete} />
        )}
        
        {quizCompleted && impactLevel && (
          <ResultsSection 
            impactLevel={impactLevel} 
            onRestart={handleRestart}
          />
        )}
        
        <section id="como-funciona" className="py-12 sm:py-16 lg:py-20 bg-card" aria-labelledby="como-funciona-title">
          <div className="container mx-auto px-4 sm:px-6 max-w-6xl">
            <h2 id="como-funciona-title" className="mb-8 sm:mb-12 text-center px-4">Como o Ethos Funciona</h2>
            <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6 sm:gap-8">
              {/* Card 1 */}
              <article className="bg-background p-6 sm:p-8 shadow-sm border border-border" style={{ borderRadius: 'calc(var(--radius) + 8px)' }}>
                <div className="mb-6">
                  <div 
                    className="w-16 h-16 rounded-full bg-primary/10 flex items-center justify-center"
                    aria-hidden="true"
                  >
                    <svg width="28" height="28" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className="text-primary">
                      <circle cx="12" cy="12" r="10"/>
                      <path d="M9.09 9a3 3 0 0 1 5.83 1c0 2-3 3-3 3"/>
                      <line x1="12" y1="17" x2="12.01" y2="17"/>
                    </svg>
                  </div>
                </div>
                <h3 className="mb-4">1. Responda às Perguntas</h3>
                <p className="text-foreground" style={{ fontSize: '16px', lineHeight: '1.6' }}>
                  O questionário apresenta situações contextuais da prática de telemedicina. 
                  Utilizamos linguagem neutra e objetiva para evitar viés.
                </p>
              </article>

              {/* Card 2 */}
              <article className="bg-background p-6 sm:p-8 shadow-sm border border-border" style={{ borderRadius: 'calc(var(--radius) + 8px)' }}>
                <div className="mb-6">
                  <div 
                    className="w-16 h-16 rounded-full bg-chart-2/10 flex items-center justify-center"
                    aria-hidden="true"
                  >
                    <svg width="28" height="28" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className="text-chart-2">
                      <path d="M17 21v-2a4 4 0 0 0-4-4H5a4 4 0 0 0-4 4v2"/>
                      <circle cx="9" cy="7" r="4"/>
                      <path d="M23 21v-2a4 4 0 0 0-3-3.87"/>
                      <path d="M16 3.13a4 4 0 0 1 0 7.75"/>
                    </svg>
                  </div>
                </div>
                <h3 className="mb-4">2. Identifique Dilemas Éticos</h3>
                <p className="text-foreground" style={{ fontSize: '16px', lineHeight: '1.6' }}>
                  Suas respostas identificam áreas onde dilemas éticos podem surgir 
                  em sua prática profissional, considerando aspectos de privacidade, segurança e responsabilidade.
                </p>
              </article>

              {/* Card 3 */}
              <article className="bg-background p-6 sm:p-8 shadow-sm border border-border" style={{ borderRadius: 'calc(var(--radius) + 8px)' }}>
                <div className="mb-6">
                  <div 
                    className="w-16 h-16 rounded-full bg-chart-3/10 flex items-center justify-center"
                    aria-hidden="true"
                  >
                    <svg width="28" height="28" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" className="text-chart-3">
                      <polyline points="22 12 18 12 15 21 9 3 6 12 2 12"/>
                    </svg>
                  </div>
                </div>
                <h3 className="mb-4">3. Avaliação de Impacto</h3>
                <p className="text-foreground" style={{ fontSize: '16px', lineHeight: '1.6' }}>
                  Ao final, você recebe uma classificação do nível de impacto social: 
                  Grave, Moderado ou Baixo, com orientações sobre aspectos éticos relevantes.
                </p>
              </article>
            </div>
          </div>
        </section>

        <section id="sobre-etica" className="py-20 bg-background" aria-labelledby="sobre-etica-title">
          <div className="container mx-auto px-6 max-w-4xl">
            <h2 id="sobre-etica-title" className="mb-8 text-center">Sobre Ética na Telemedicina</h2>
            <div className="space-y-6">
              <p style={{ fontSize: '16px', lineHeight: '1.7' }}>
                A telemedicina transformou o acesso aos cuidados de saúde, trazendo novos 
                desafios éticos que profissionais de tecnologia devem considerar ao desenvolver 
                soluções nesta área.
              </p>
              <p style={{ fontSize: '16px', lineHeight: '1.7' }}>
                Questões de sigilo médico, segurança de dados, consentimento informado 
                e equidade no acesso são fundamentais para garantir que a tecnologia 
                beneficie pacientes e profissionais de saúde.
              </p>
              <div 
                className="p-6 mt-8" 
                style={{ 
                  backgroundColor: '#fef3c7',
                  borderRadius: 'calc(var(--radius) + 8px)' 
                }}
                role="note"
                aria-labelledby="disclaimer-title"
              >
                <h3 id="disclaimer-title" className="mb-4">Declaração de Responsabilidade</h3>
                <p className="text-foreground" style={{ fontSize: '16px', lineHeight: '1.6' }}>
                  Este simulador é uma ferramenta educacional e não substitui orientação profissional 
                  especializada em ética médica ou jurídica. Os resultados são baseados 
                  em suas respostas e servem como ponto de reflexão inicial sobre dilemas éticos.
                </p>
              </div>
            </div>
          </div>
        </section>

        <section id="contato" className="py-20 bg-card" aria-labelledby="contato-title">
          <div className="container mx-auto px-6 max-w-2xl text-center">
            <h2 id="contato-title" className="mb-6">Entre em Contato</h2>
            <p className="mb-8 text-foreground" style={{ fontSize: '16px', lineHeight: '1.6' }}>
              Possui dúvidas ou sugestões sobre o Ethos? Nossa equipe está disponível para auxiliar.
            </p>
            <a 
              href="mailto:contato@ethos.example.com"
              className="inline-block px-8 py-4 bg-primary text-primary-foreground hover:opacity-90 transition-opacity focus:outline-none focus:ring-2 focus:ring-ring focus:ring-offset-4"
              style={{ 
                borderRadius: 'var(--radius)', 
                fontSize: '16px',
                fontWeight: 'var(--font-weight-bold)'
              }}
              aria-label="Enviar mensagem para contato@ethos.example.com"
            >
              Enviar Mensagem
            </a>
          </div>
        </section>
      </main>

      <Footer />
      
      {/* Floating Feedback Button */}
      <FloatingFeedbackButton onClick={handleOpenFeedback} />
      
      {/* Feedback Modal */}
      {showFeedbackModal && (
        <FeedbackModal 
          questionNumber={0} 
          onClose={handleCloseFeedback}
        />
      )}
    </div>
  );
}
